23  // Writes the bytes of a value of arbitrary size in guest order
24  template <typename Value_t>
25  void write_bytes(std::stringstream &ss, Value_t value) {
26  unsigned char *p = (unsigned char*)&value;
27  unsigned char *end = p + sizeof(Value_t);
28 
29  ss << std::hex << std::setfill('0');
30  while (p != end)
31  ss << std::setw(2) << (unsigned)(*p++);
32  }
33 
34  void write_byte(std::stringstream &ss, uint8_t byte) {
35  ss << std::hex << std::setfill('0');
36  ss << std::setw(2) << (unsigned)byte;
37  }

39  std::string hexify(const std::string& s) {
40  std::stringstream ss;
41  ss << std::hex << std::setfill('0');
42  for (const unsigned char &c : s)
43  ss << std::setw(2) << (unsigned)c;
44  return ss.str();
45  }

192  std::visit(util::overloaded {
193  [&ss](const xd::reg::x86_64::RegistersX86_64& regs) {

206  }
