157  uint8_t read_byte() {
158  const auto from_hex = [](const auto c) {
159  if (c >= '0' && c <= '9')
160  return c - '0';
161 
162  auto cl = std::tolower(c);
163  if (cl >= 'a' && cl <= 'f')
164  return 0xa + (cl - 'a');
165 
166  throw RequestPacketParseException();
167  };
168 
169  uint8_t c1 = from_hex(get_char());
170  uint8_t c2 = from_hex(get_char());
171 
172  return (c1 << 4) + c2;
173  };

186  template <typename Value_t>
187  Value_t read_hex_number_respecting_endianness() {
188  Value_t value;
189  uint8_t *value_ptr = (uint8_t*)&value;
190  size_t remaining = 2*sizeof(Value_t);
191 
192  while (has_more() && remaining) {
193  *value_ptr++ = read_byte();
194  remaining -= 2;
195  }
196 
197  if (remaining)
198  throw RequestPacketParseException();

200  return value;

225  static constexpr auto SIZE = 2*sizeof(Word_t);
226  static const auto NOT_SPECIFIED = std::string(SIZE, 'x');
